<?php 
	header('content-type:text/html; charset=utf-8');
	//@符號可抑制錯誤輸出,可壁面無法讀取檔案時,
	//PHP輸出錯誤訊息,影響網頁版面與畫面
	if(@$counter = file_get_contents("test01.php"))
	{
		echo 'test01.php的內容如下<hr>' .

			 	//把一些預定義的字符轉換為HTML實體
			 	htmlspecialchars($counter) . 
			 '</pre>';

	}
	else
		echo '讀取失敗';
?>
